204 template <typename Array_Type, typename Data_Counter_Type, typename Data_Rule_Type>
205 inline void Support<Array_Type, Data_Counter_Type, Data_Rule_Type>::calc_backend(
206  uint pos,
207  std::vector< Array_Type > * array_bank,
208  std::vector< std::vector< double > > * stats_bank
209  ) {
210 
211  // Did we reached the end??
212  if (pos >= coordinates_free.size())
213  return;
214 
215  // We will pass it to the next step, if the iteration makes sense.
216  calc_backend(pos + 1u, array_bank, stats_bank);
217 
218  // Once we have returned, everything will be back as it used to be, so we
219  // treat the data as if nothing has changed.
220 
221  // Toggle the cell (we will toggle it back after calling the counter)
222  EmptyArray.insert_cell(
223  coordinates_free[pos].first,
224  coordinates_free[pos].second,
225  EmptyArray.Cell_default , false, false
226  );
227 
228  // Counting
229  // std::vector< double > change_stats(counters.size());
230  for (uint n = 0u; n < counters->size(); ++n) {
231  change_stats[pos][n] = counters->operator[](n)->count(
232  &EmptyArray,
233  coordinates_free[pos].first,
234  coordinates_free[pos].second
235  );
236  current_stats[n] += change_stats[pos][n];
237  }
238 
239  // Adding to the overall count
240  data.add(current_stats);
241 
242  // Need to save?
243  if (array_bank != nullptr)
244  array_bank->push_back(EmptyArray);
245 
246  if (stats_bank != nullptr)
247  stats_bank->push_back(current_stats);
248 
249  // Again, we only pass it to the next level iff the next level is not
250  // passed the last step.
251  calc_backend(pos + 1, array_bank, stats_bank);
252 
253  // We need to restore the state of the cell
254  EmptyArray.rm_cell(
255  coordinates_free[pos].first,
256  coordinates_free[pos].second,
257  false, false
258  );
259 
260  for (uint n = 0u; n < counters->size(); ++n)
261  current_stats[n] -= change_stats[pos][n];
262 
263 
264  return;
265 
266 }
